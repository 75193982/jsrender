Providing a custom tag

{{#sort Languages reverse=true}}...{{/sort}}

$.views.registerTags({

    sort: function( array ){
        var ret = "";
        if ( this.reverse ) {
            // Render in reverse order
            for ( var l = array.length, i = l; i; i-- ) {
                ret += $.render( array[ i - 1 ], this.tmpl );
            }
        } else {
            // Render in original order
            ret += $.render( array, this.tmpl );
        }
        return ret;
    }

});
